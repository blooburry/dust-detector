Table User {
 id int [pk, increment]
 username varchar(255) [not null]
 password varchar(255) [not null]
 phone varchar(20)
 email varchar(255)
 role int [not null]
 verificationKey varchar(255)
 indexes {
   (email) [unique]
 }
}

Table Detector {
 id int [pk, increment]
 ip_address varchar(15) [not null]
 api_key varchar(255) [not null]
 dateBought date [not null]
 status int [not null]
 user_id int [not null]
 indexes {
   (ip_address) [unique]
   (user_id)
 }
}

Table DustLevel {
 detectorId int [not null]
 level int [not null]
 date date [not null]
 indexes {
   (detectorId, date) [pk]
 }
}

Ref: Detector.user_id > User.id
Ref: DustLevel.detectorId > Detector.id
